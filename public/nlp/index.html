💡 Text-Klassifikation mit spaCy

import spacy
from spacy.pipeline.textcat import Config

nlp = spacy.blank("de")

# Named Entity Recognition
nlp = spacy.load("de_core_news_sm")
doc = nlp("Angela Merkel war in Berlin.")

for ent in doc.ents:
    print(f"{ent.text:20} → {ent.label_}")
# Angela Merkel → PER
# Berlin        → LOC
