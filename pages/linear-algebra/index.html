💡 Erste Schritte in NumPy

import numpy as np

v = np.array([1, 2, 3])          # Vektor
A = np.array([[1,2],[3,4]])       # Matrix
print(A @ v[:2])                  # Matrix-Vektor-Multiplikation
print(np.dot(v, v))               # Skalarprodukt = 14
print(A.T)                        # Transponierung

💡 Eigenwerte mit NumPy

import numpy as np

A = np.array([[2, 1], [1, 2]])
eigenvalues, eigenvectors = np.linalg.eig(A)
print(eigenvalues)    # [3. 1.]
print(eigenvectors)   # Spalten sind Eigenvektoren

# Gleichungssystem Ax = b
b = np.array([1, 2])
x = np.linalg.solve(A, b)
print(x)

  • Ich kenne den Zusammenhang zwischen Determinante und Invertierbarkeit
  • Ich kann ein 3×3-Gleichungssystem mit Gauß lösen

💡 SVD und PCA

import numpy as np

# SVD
A = np.random.randn(100, 20)
U, sigma, Vt = np.linalg.svd(A, full_matrices=False)

# PCA via SVD (auf zentrierten Daten)
X = A - A.mean(axis=0)
_, _, Vt = np.linalg.svd(X, full_matrices=False)
k = 5  # Top-k Hauptkomponenten
X_reduced = X @ Vt[:k].T   # (100, 5)
